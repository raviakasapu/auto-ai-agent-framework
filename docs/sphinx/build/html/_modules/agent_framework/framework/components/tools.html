    def execute(
        self,
        model_dir: str,
        from_table: str,
        from_column: str,
        to_table: str,
        to_column: str,
        is_active: bool = True,
        from_cardinality: str | None = None,
        cross_filtering_behavior: str | None = None,
    ) -> dict:
        root = Path(model_dir)
        if not root.exists():
            raise FileNotFoundError(f"model_dir not found: {model_dir}")
        tables_dir = root / "Model" / "tables"
        rel_file = root / "Model" / "relationships.tmdl"
        # Validate endpoints
        self._ensure_column_exists(tables_dir, from_table, from_column)
        self._ensure_column_exists(tables_dir, to_table, to_column)

        rel_id = uuid.uuid4().hex
        lines = [
            f"relationship {rel_id}",
            f"  fromColumn: {from_table}.{from_column}",
            f"  toColumn: {to_table}.{to_column}",
        ]
        if cross_filtering_behavior:
            lines.append(f"  crossFilteringBehavior: {cross_filtering_behavior}")
        if from_cardinality:
            lines.append(f"  fromCardinality: {from_cardinality}")
        lines.append(f"  isActive: {'true' if is_active else 'false'}")
        block = "\n".join(lines) + "\n\n"

        # Append (or create) relationships file
        rel_file.parent.mkdir(parents=True, exist_ok=True)
        if rel_file.exists():
            current = rel_file.read_text(encoding="utf-8")
            if current and not current.endswith("\n\n"):
                block = "\n" + block
            rel_file.write_text(current + block, encoding="utf-8")
        else:
            rel_file.write_text(block, encoding="utf-8")

        out = AddRelationshipOutput(
            id=rel_id,
            relationships_file=str(rel_file),
            message="Relationship appended",
        )
        return out.model_dump()



    def _ensure_column_exists(self, tables_dir: Path, table: str, column: str) -> None:
        # Find file by name under tables_dir (case-insensitive)
        candidate = tables_dir / f"{table}.tmdl"
        if not candidate.exists():
            # try case-insensitive
            matches = [p for p in tables_dir.glob("*.tmdl") if p.stem.lower() == table.lower()]
            if matches:
                candidate = matches[0]
            else:
                raise FileNotFoundError(f"Table TMDL not found for '{table}' in {tables_dir}")
        text = candidate.read_text(encoding="utf-8")
        cols = self._parse_columns(text)
        if column not in cols:
            # try case-insensitive
            if column.lower() not in {c.lower() for c in cols}:
                raise ValueError(f"Column '{column}' not found in table '{table}'")

    def _parse_columns(self, content: str) -> set[str]:
        cols: set[str] = set()
        # Matches: column 'Name' or column "Name"
        for m in re.finditer(r"^\s*column\s+['\"]([^'\"]+)['\"]", content, flags=re.MULTILINE | re.IGNORECASE):
            cols.add(m.group(1))
        return cols

    def execute(
        self,
        model_dir: str,
        id: str,
        from_table: str | None = None,
        from_column: str | None = None,
        to_table: str | None = None,
        to_column: str | None = None,
        is_active: bool | None = None,
        from_cardinality: str | None = None,
        cross_filtering_behavior: str | None = None,
    ) -> dict:
        root = Path(model_dir)
        rel_file = root / "Model" / "relationships.tmdl"
        if not rel_file.exists():
            raise FileNotFoundError(f"relationships.tmdl not found under {rel_file}")
        text = rel_file.read_text(encoding="utf-8")
        blocks = self._split_blocks(text)
        updated = False
        for i, blk in enumerate(blocks):
            head = blk.splitlines()[0].strip() if blk.strip() else ""
            if head.startswith("relationship ") and head.split(" ")[1] == id:
                blk_lines = blk.splitlines()
                blk_lines = self._patch_field(blk_lines, "fromColumn:", f"{from_table}.{from_column}" if from_table and from_column else None)
                blk_lines = self._patch_field(blk_lines, "toColumn:", f"{to_table}.{to_column}" if to_table and to_column else None)
                blk_lines = self._patch_field(blk_lines, "isActive:", "true" if is_active is True else "false" if is_active is not None else None)
                blk_lines = self._patch_field(blk_lines, "fromCardinality:", from_cardinality)
                blk_lines = self._patch_field(blk_lines, "crossFilteringBehavior:", cross_filtering_behavior)
                blocks[i] = "\n".join(blk_lines)
                updated = True
                break
        if not updated:
            raise ValueError(f"Relationship id not found: {id}")
        new_text = ("\n\n").join([b for b in blocks if b.strip()]) + "\n"
        rel_file.write_text(new_text, encoding="utf-8")
        return UpdateRelationshipOutput(id=id, relationships_file=str(rel_file), message="Relationship updated").model_dump()



    def _split_blocks(self, content: str) -> list[str]:
        # Split on blank line boundaries between relationship blocks
        parts: list[str] = []
        buf: list[str] = []
        for line in content.splitlines():
            if not line.strip():
                if buf:
                    parts.append("\n".join(buf))
                    buf = []
            else:
                buf.append(line)
        if buf:
            parts.append("\n".join(buf))
        return parts

    def _patch_field(self, lines: list[str], key: str, value: str | None) -> list[str]:
        if value is None:
            return lines
        found = False
        for i, line in enumerate(lines):
            if line.strip().startswith(key):
                lines[i] = re.sub(rf"^\s*{re.escape(key)}.*$", f"  {key} {value}", lines[i])
                found = True
                break
        if not found:
            lines.append(f"  {key} {value}")
        return lines

    def execute(self, file_path: str, content: str, write_mode: str = "overwrite") -> dict:
        p = Path(file_path)
        p.parent.mkdir(parents=True, exist_ok=True)
        if write_mode == "append" and p.exists():
            old = p.read_text(encoding="utf-8")
            p.write_text(old + content, encoding="utf-8")
        else:
            p.write_text(content, encoding="utf-8")
        return TMDLWriterOutput(file_path=file_path, message="Write complete").model_dump()
