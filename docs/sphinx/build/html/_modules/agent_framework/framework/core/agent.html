from __future__ import annotations

from typing import Any, Dict, List, Optional

from .events import EventBus
from pydantic import ValidationError

class Agent:
    def __init__(
        self,
        planner: BasePlanner,
        memory: BaseMemory,
        tools: List[BaseTool] | Dict[str, BaseTool],
        event_bus: Optional[EventBus] = None,
        name: str | None = None,
        description: str | None = None,
        version: str | None = None,
    ) -> None:
        self.planner = planner
        self.memory = memory
        # Normalize tools to a dict by name for lookup
        if isinstance(tools, dict):
            self.tools: Dict[str, BaseTool] = tools
        else:
            self.tools = {t.name: t for t in tools}
        self.event_bus = event_bus or EventBus()
        self.name = name or "Agent"
        self.description = description or "An AI agent."
        self.version = version or "0.1.0"

    async def run(self, task: str, progress_handler: Optional[BaseProgressHandler] = None) -> Any:
        # Start event
        start_data = {"task": task}
        self.event_bus.publish("agent_start", start_data)
        if progress_handler:
            await progress_handler.on_event("agent_start", start_data)

        self.memory.add({"type": "task", "content": task})

        # Plan
        plan_outcome = self.planner.plan(task, self.memory.get_history())
        if isinstance(plan_outcome, Action):
            planned_data = {"tool": plan_outcome.tool_name, "args": plan_outcome.tool_args}
            self.event_bus.publish("action_planned", planned_data)
            if progress_handler:
                await progress_handler.on_event("action_planned", planned_data)

            tool = self.tools.get(plan_outcome.tool_name)
            if not tool:
                msg = f"Tool not found: {plan_outcome.tool_name}"
                err_data = {"message": msg}
                self.event_bus.publish("error", err_data)
                if progress_handler:
                    await progress_handler.on_event("error", err_data)
                self.memory.add({"type": "error", "content": msg})
                self.event_bus.publish("agent_end", {"result": msg})
                if progress_handler:
                    await progress_handler.on_event("agent_end", {"result": msg})
                return msg

            # Validate and normalize tool args using the tool's Pydantic schema
            tool_kwargs: Dict[str, Any] = plan_outcome.tool_args
            if getattr(tool, "args_schema", None):
                try:
                    model = tool.args_schema.model_validate(tool_kwargs)
                    tool_kwargs = model.model_dump()
                except ValidationError as ve:
                    msg = f"Tool argument validation failed for {tool.name}: {ve}"
                    err_data = {"message": msg}
                    self.event_bus.publish("error", err_data)
                    if progress_handler:
                        await progress_handler.on_event("error", err_data)
                    self.memory.add({"type": "error", "content": msg})
                    self.event_bus.publish("agent_end", {"result": msg})
                    if progress_handler:
                        await progress_handler.on_event("agent_end", {"result": msg})
                    return msg

            result = tool.execute(**tool_kwargs)
            executed_data = {"tool": tool.name, "result": result}
            self.event_bus.publish("action_executed", executed_data)
            if progress_handler:
                await progress_handler.on_event("action_executed", executed_data)

            # Record to memory
            self.memory.add({"type": "action", "tool": tool.name, "args": plan_outcome.tool_args})
            self.memory.add({"type": "observation", "content": result})

            # In this simple loop, the first action's result is the final answer
            end_data = {"result": result}
            self.event_bus.publish("agent_end", end_data)
            if progress_handler:
                await progress_handler.on_event("agent_end", end_data)
            return result
        else:
            # Final response came directly from planner
            final = plan_outcome.content
            self.memory.add({"type": "final", "content": final})
            end_data = {"result": final}
            self.event_bus.publish("agent_end", end_data)
            if progress_handler:
                await progress_handler.on_event("agent_end", end_data)
            return final
