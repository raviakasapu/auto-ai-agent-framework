class ManagerAgent:
    """A minimal manager agent that delegates to worker agents.

    This simple implementation delegates the task to the first worker and returns
    its result. It publishes events and supports async progress callbacks similar
    to the standard Agent.
    """

    def __init__(
        self,
        planner: BasePlanner,
        memory: BaseMemory,
        workers: List[Any],  # list of Agent-like objects with async run()
        event_bus: Optional[EventBus] = None,
        name: str | None = None,
        description: str | None = None,
        version: str | None = None,
    ) -> None:
        self.planner = planner
        self.memory = memory
        self.workers = workers
        self.event_bus = event_bus or EventBus()
        self.name = name or "ManagerAgent"
        self.description = description or "An AI manager agent."
        self.version = version or "0.1.0"

    async def run(self, task: str, progress_handler: Optional[BaseProgressHandler] = None) -> Any:
        start_data = {"task": task, "workers": len(self.workers)}
        self.event_bus.publish("manager_start", start_data)
        if progress_handler:
            await progress_handler.on_event("manager_start", start_data)

        self.memory.add({"type": "task", "content": task})

        # For now, simply choose the first worker
        if not self.workers:
            msg = "No workers available for delegation"
            self.event_bus.publish("error", {"message": msg})
            if progress_handler:
                await progress_handler.on_event("error", {"message": msg})
            self.event_bus.publish("manager_end", {"result": msg})
            if progress_handler:
                await progress_handler.on_event("manager_end", {"result": msg})
            return msg

        chosen = self.workers[0]
        self.event_bus.publish("delegation_chosen", {"worker": getattr(chosen, "name", "worker0")})
        if progress_handler:
            await progress_handler.on_event("delegation_chosen", {"worker": getattr(chosen, "name", "worker0")})

        result = await chosen.run(task, progress_handler=progress_handler)
        self.memory.add({"type": "observation", "content": result})

        end_data = {"result": result}
        self.event_bus.publish("manager_end", end_data)
        if progress_handler:
            await progress_handler.on_event("manager_end", end_data)
        return result
