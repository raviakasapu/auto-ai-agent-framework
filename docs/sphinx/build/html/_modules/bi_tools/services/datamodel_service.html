Central Data Model Service - Internal API

Simple service for reading/writing to the central data model JSON.
This is the single source of truth for all model operations.

Usage::

    from bi_tools.services.datamodel_service import DataModelService

    service = DataModelService()

    # Read operations
    model = service.get_model("my_model_id")
    tables = service.get_tables("my_model_id")
    table = service.get_table("my_model_id", "ORDER_DETAIL")

    # Write operations
    service.add_column("my_model_id", "ORDER_DETAIL", {
        "name": "NewColumn",
        "dataType": "string"
    })
    service.add_measure("my_model_id", "ORDER_DETAIL", {
        "name": "TotalSales",
        "expression": "SUM(...)"
    })

# IMPORTANT: Use contextvars instead of threading.local() for async/await compatibility
# This ensures proper isolation across concurrent async tasks (WebSocket connections, etc.)
_current_job_id: contextvars.ContextVar[Optional[str]] = contextvars.ContextVar(
    '_current_job_id',
    default=None
)


def _get_current_job_id() -> Optional[str]:
    """Get the job_id from context-local storage (async-safe)"""
    return _current_job_id.get()


def _set_current_job_id(job_id: str) -> None:
    """Set the job_id in context-local storage (async-safe)"""
    _current_job_id.set(job_id)


def _clear_current_job_id() -> None:
    """Clear the job_id from context-local storage (async-safe)"""
    _current_job_id.set(None)

@contextmanager
def datamodel_context(job_id: str):
    """
    Context manager for request-scoped data model operations.
    
    This ensures each request (user/job) gets isolated access to their data model,
    even when multiple requests are processed concurrently.
    
    Usage::

        with datamodel_context(job_id="abc123"):
            service = get_datamodel_service()  # Automatically uses job_id="abc123"
            agent.run(task)  # All tools use the correct model
    
    Async-Safety:
        Uses contextvars (not threading.local!) for proper isolation across
        concurrent async tasks in the same thread (e.g., multiple WebSocket connections).
    """
    previous_job_id = _get_current_job_id()
    try:
        _set_current_job_id(job_id)
        logger.debug(f"Entered datamodel context for job_id: {job_id}")
        yield
    finally:
        if previous_job_id:
            _set_current_job_id(previous_job_id)
        else:
            _clear_current_job_id()
        logger.debug(f"Exited datamodel context for job_id: {job_id}")

    def __init__(self, job_id: Optional[str] = None, base_dir: str = "central_data_models"):
        """
        Initialize DataModelService.
        
        Args:
            job_id: Job ID for this data model (optional - will use MODEL_ID from env)
            base_dir: Base directory for data models
        """
        # Get job_id from parameter or environment
        resolved_job_id = job_id or os.getenv("MODEL_ID") or os.getenv("JOB_ID")
        
        if not resolved_job_id:
            # Fallback: look for any existing job directories
            base_path = Path(base_dir)
            if base_path.exists():
                job_dirs = [d for d in base_path.iterdir() if d.is_dir() and (d / "datamodel.json").exists()]
                if job_dirs:
                    resolved_job_id = job_dirs[0].name
                else:
                    # Check legacy format
                    legacy_path = base_path / "datamodel.json"
                    if legacy_path.exists():
                        # Use legacy single-file format
                        resolved_job_id = "legacy"
                        self.job_id = resolved_job_id
                        self.datamodel_path = legacy_path
                        self._lock = RLock()
                        self._cache = {}
                        self._is_nested_format = False
                        self._wrapper_data = {}
                        self._load_datamodel()
                        return
        
        if not resolved_job_id:
            raise ValueError(
                "No job_id available. Set MODEL_ID or JOB_ID environment variable."
            )
        
        # Set job_id after validation
        self.job_id = resolved_job_id
        
        # NEW structure: central_data_models/{job_id}/datamodel.json
        self.datamodel_path = Path(base_dir) / self.job_id / "datamodel.json"
        self._lock = RLock()  # Re-entrant lock for nested operations
        self._cache = {}
        self._is_nested_format = False  # Track if loaded from nested format
        self._wrapper_data = {}
        self._load_datamodel()


    
    def _load_datamodel(self) -> None:
        """Load data model from disk into memory"""
        with self._lock:
            if not self.datamodel_path.exists():
                logger.warning(f"Data model not found: {self.datamodel_path}")
                # Initialize empty model structure
                self._cache = {
                    "meta": {"jobId": self.job_id},
                    "tables": [],
                    "relationships": []
                }
                self._is_nested_format = False
                self._wrapper_data = {}
                return
            
            try:
                with open(self.datamodel_path, 'r', encoding='utf-8') as f:
                    data = json.load(f)
                
                # Handle two formats:
                # 1. Flat format: { "meta": {...}, "tables": [...], "relationships": [...] }
                # 2. Nested format: { "model": { "meta": {...}, "tables": [...], ... } }
                if "model" in data and isinstance(data["model"], dict):
                    # Nested format - extract the model object
                    logger.info(f"Detected nested model format, extracting model data")
                    self._is_nested_format = True
                    self._cache = data["model"]
                    # Store the wrapper data for preservation
                    self._wrapper_data = {k: v for k, v in data.items() if k != "model"}
                else:
                    # Flat format
                    self._is_nested_format = False
                    self._cache = data
                    self._wrapper_data = {}
                
                logger.info(f"Loaded data model for job {self.job_id} from {self.datamodel_path}")
            except Exception as e:
                logger.error(f"Failed to load data model: {e}")
                self._cache = {
                    "meta": {"jobId": self.job_id},
                    "tables": [],
                    "relationships": []
                }
                self._is_nested_format = False
                self._wrapper_data = {}
    
    def _save_datamodel(self) -> None:
        """Save data model from memory to disk"""
        with self._lock:
            try:
                # Create job-specific directory
                self.datamodel_path.parent.mkdir(parents=True, exist_ok=True)
                
                # Ensure metadata
                if "meta" not in self._cache:
                    self._cache["meta"] = {}
                self._cache["meta"]["jobId"] = self.job_id
                self._cache["meta"]["lastUpdated"] = datetime.utcnow().isoformat()
                
                # Preserve original format when saving
                if self._is_nested_format:
                    # Wrap in nested format
                    save_data = {
                        "model": self._cache,
                        **self._wrapper_data  # Preserve model_source, model_version, etc.
                    }
                else:
                    # Flat format
                    save_data = self._cache
                
                with open(self.datamodel_path, 'w', encoding='utf-8') as f:
                    json.dump(save_data, f, indent=2, ensure_ascii=False)
                
                logger.info(f"Saved data model for job {self.job_id} to {self.datamodel_path}")
            except Exception as e:
                logger.error(f"Failed to save data model: {e}")
                raise

    def add_table(self, table_data: Dict[str, Any]) -> bool:
        """Add a new table to the current model"""
        with self._lock:
            # Check if table already exists
            table_name = table_data.get("name")
            if not table_name or not isinstance(table_name, str):
                logger.error("Table data must include a 'name' field")
                return False
            if self.get_table(table_name):
                logger.error(f"Table already exists: {table_name}")
                return False
            
            # Ensure required fields
            if "columns" not in table_data:
                table_data["columns"] = []
            if "measures" not in table_data:
                table_data["measures"] = []
            
            if "tables" not in self._cache:
                self._cache["tables"] = []
            
            self._cache["tables"].append(table_data)
            self._save_datamodel()
            logger.info(f"Added table: {table_name}")
            return True

    def add_column(self, table_name: str, column_data: Dict[str, Any]) -> bool:
        """Add a column to a table"""
        with self._lock:
            table = self.get_table(table_name)
            if not table:
                logger.error(f"Table not found: {table_name}")
                return False
            
            column_name = column_data.get("name")
            if not column_name or not isinstance(column_name, str):
                logger.error("Column data must include a 'name' field")
                return False
            
            # Check if column already exists
            if self.get_column(table_name, column_name):
                logger.error(f"Column already exists: {column_name}")
                return False
            
            # Ensure required fields
            if "dataType" not in column_data:
                column_data["dataType"] = "string"
            
            table["columns"].append(column_data)
            self._save_datamodel()
            logger.info(f"Added column: {column_name} to table {table_name}")
            return True

    def update_column(self, table_name: str, column_name: str, updates: Dict[str, Any]) -> bool:

                    return True
            
            logger.error(f"Column not found: {column_name}")
            return False

    def delete_column(self, table_name: str, column_name: str) -> bool:
        """Delete a column from a table"""
        with self._lock:
            table = self.get_table(table_name)
            if not table:
                logger.error(f"Table not found: {table_name}")
                return False
            
            columns = table.get("columns", [])
            original_count = len(columns)
            table["columns"] = [c for c in columns if c.get("name") != column_name]
            
            if len(table["columns"]) == original_count:
                logger.error(f"Column not found: {column_name}")
                return False
            
            self._save_datamodel()
            logger.info(f"Deleted column: {column_name} from table {table_name}")
            return True

    def get_rich_context(self, max_tables: int = 100, max_columns_per_table: int = 20, max_measures_per_table: int = 10, format: str = "json") -> Dict[str, Any]:
        """
        Get a rich, structured context for orchestrator/manager planning.
        
        This provides a comprehensive but compact view of the data model that enables
        better decision-making without requiring multiple tool calls.
        
        Args:
            max_tables: Maximum number of tables to include (default 100)
            max_columns_per_table: Maximum columns per table to show (default 20, shows key columns first)
            max_measures_per_table: Maximum measures per table to show (default 10)
            format: Output format - "json" (nested structure) or "graph" (knowledge graph format)
        
        Returns:
            Dictionary with structured model information:
            - If format="json": nested structure with summary, tables, relationships, dimensions, facts
            - If format="graph": knowledge graph format with nodes and edges
        """
        tables = self._cache.get("tables", [])
        relationships = self._cache.get("relationships", [])
        
        # Build relationship index for quick lookup
        rel_index = {}
        for rel in relationships:
            from_table = (rel.get('from') or {}).get('table') or rel.get('fromTable')
            from_col = (rel.get('from') or {}).get('column') or rel.get('fromColumn') or rel.get('fromColumnName')
            to_table = (rel.get('to') or {}).get('table') or rel.get('toTable')
            to_col = (rel.get('to') or {}).get('column') or rel.get('toColumn') or rel.get('toColumnName')
            
            if from_table and to_table:
                key = f"{from_table}.{from_col}"
                if key not in rel_index:
                    rel_index[key] = []
                rel_index[key].append({
                    "to": f"{to_table}.{to_col}",
                    "active": rel.get("isActive", True),
                    "cardinality": rel.get("fromCardinality", "many")
                })
        
        # Identify key columns (those used in relationships)
        key_columns = set()
        for rel in relationships:
            from_col = (rel.get('from') or {}).get('column') or rel.get('fromColumn') or rel.get('fromColumnName')
            to_col = (rel.get('to') or {}).get('column') or rel.get('toColumn') or rel.get('toColumnName')
            from_table = (rel.get('from') or {}).get('table') or rel.get('fromTable')
            to_table = (rel.get('to') or {}).get('table') or rel.get('toTable')
            if from_table and from_col:
                key_columns.add(f"{from_table}.{from_col}")
            if to_table and to_col:
                key_columns.add(f"{to_table}.{to_col}")
        
        # Classify tables as fact or dimension
        def is_dimension(table_name: str) -> bool:
            name_lower = table_name.lower()
            return (name_lower.startswith('dim') or 
                    name_lower.startswith('lu_') or 
                    'lookup' in name_lower or 
                    name_lower in {'date', 'calendar', 'time'})
        
        # Build table summaries
        table_summaries = []
        for table in tables[:max_tables]:
            table_name = table.get("name", "")
            columns = table.get("columns", [])
            measures = table.get("measures", [])
            
            # Prioritize key columns (those used in relationships)
            key_cols = []
            other_cols = []
            for col in columns:
                col_name = col.get("name", "")
                col_key = f"{table_name}.{col_name}"
                if col_key in key_columns:
                    key_cols.append({
                        "name": col_name,
                        "dataType": col.get("dataType", "unknown"),
                        "isKey": True
                    })
                else:
                    other_cols.append({
                        "name": col_name,
                        "dataType": col.get("dataType", "unknown")
                    })
            
            # Combine: key columns first, then others (limited)
            all_cols = key_cols + other_cols[:max(0, max_columns_per_table - len(key_cols))]
            
            # Measure summaries
            measure_summaries = [
                {
                    "name": m.get("name", ""),
                    "table": table_name
                }
                for m in measures[:max_measures_per_table]
            ]
            
            table_summaries.append({
                "name": table_name,
                "type": "dimension" if is_dimension(table_name) else "fact",
                "columnCount": len(columns),
                "measureCount": len(measures),
                "keyColumns": [c["name"] for c in key_cols],
                "columns": [c["name"] for c in all_cols],  # All shown columns
                "measures": [m["name"] for m in measure_summaries],
                "hasMoreColumns": len(columns) > len(all_cols),
                "hasMoreMeasures": len(measures) > len(measure_summaries)
            })
        
        # Build relationship graph (compact format)
        rel_graph = []
        for rel in relationships[:200]:  # Cap at 200 relationships
            from_table = (rel.get('from') or {}).get('table') or rel.get('fromTable')
            from_col = (rel.get('from') or {}).get('column') or rel.get('fromColumn') or rel.get('fromColumnName')
            to_table = (rel.get('to') or {}).get('table') or rel.get('toTable')
            to_col = (rel.get('to') or {}).get('column') or rel.get('toColumn') or rel.get('toColumnName')
            
            if from_table and to_table and from_col and to_col:
                rel_graph.append({
                    "from": f"{from_table}.{from_col}",
                    "to": f"{to_table}.{to_col}",
                    "active": rel.get("isActive", True),
                    "cardinality": rel.get("fromCardinality", "many")
                })
        
        # Separate dimensions and facts
        dimensions = [t["name"] for t in table_summaries if t["type"] == "dimension"]
        facts = [t["name"] for t in table_summaries if t["type"] == "fact"]
        
        if format == "graph":
            # Knowledge graph representation - more compact and easier for LLMs to reason about
            return self._build_knowledge_graph(table_summaries, rel_graph, dimensions, facts, len(tables), len(relationships))
        else:
            # Traditional JSON nested structure
            return {
                "summary": {
                    "tableCount": len(tables),
                    "relationshipCount": len(relationships),
                    "dimensionCount": len(dimensions),
                    "factCount": len(facts),
                    "totalColumns": sum(t.get("columnCount", 0) for t in table_summaries),
                    "totalMeasures": sum(t.get("measureCount", 0) for t in table_summaries)
                },
                "tables": table_summaries,
                "relationships": rel_graph,
                "dimensions": dimensions,
                "facts": facts
            }


    
    def _build_knowledge_graph(self, table_summaries: List[Dict], rel_graph: List[Dict], 
                               dimensions: List[str], facts: List[str], 
                               total_tables: int, total_relationships: int) -> Dict[str, Any]:
        """
        Build a knowledge graph representation of the data model.
        
        Format:
        {
            "summary": {...},
            "nodes": [
                {"id": "TableName", "type": "table|column|measure", "properties": {...}},
                ...
            ],
            "edges": [
                {"from": "Table1.Column1", "to": "Table2.Column2", "type": "relationship", "properties": {...}},
                {"from": "Table1", "to": "Table1.Column1", "type": "has_column"},
                {"from": "Table1", "to": "Table1.Measure1", "type": "has_measure"},
                ...
            ]
        }
        """
        nodes = []
        edges = []
        
        # Add table nodes
        for table in table_summaries:
            table_name = table["name"]
            nodes.append({
                "id": table_name,
                "type": "table",
                "properties": {
                    "type": table["type"],  # dimension or fact
                    "columnCount": table["columnCount"],
                    "measureCount": table["measureCount"],
                    "keyColumns": table["keyColumns"]
                }
            })
            
            # Add column nodes and edges (only key columns + sample others)
            for col_name in table["columns"]:
                col_id = f"{table_name}.{col_name}"
                is_key = col_name in table["keyColumns"]
                
                # Add column node (only if key column to reduce size)
                if is_key:
                    nodes.append({
                        "id": col_id,
                        "type": "column",
                        "properties": {
                            "table": table_name,
                            "isKey": True
                        }
                    })
                    # Edge: table has column
                    edges.append({
                        "from": table_name,
                        "to": col_id,
                        "type": "has_column",
                        "properties": {"isKey": True}
                    })
                else:
                    # For non-key columns, just add edge without node (more compact)
                    edges.append({
                        "from": table_name,
                        "to": col_id,
                        "type": "has_column",
                        "properties": {"isKey": False, "name": col_name}
                    })
            
            # Add measure nodes and edges
            for measure_name in table["measures"]:
                measure_id = f"{table_name}.{measure_name}"
                nodes.append({
                    "id": measure_id,
                    "type": "measure",
                    "properties": {
                        "table": table_name
                    }
                })
                # Edge: table has measure
                edges.append({
                    "from": table_name,
                    "to": measure_id,
                    "type": "has_measure"
                })
        
        # Add relationship edges
        for rel in rel_graph:
            edges.append({
                "from": rel["from"],
                "to": rel["to"],
                "type": "relationship",
                "properties": {
                    "active": rel["active"],
                    "cardinality": rel.get("cardinality", "many")
                }
            })
        
        return {
            "format": "knowledge_graph",
            "summary": {
                "tableCount": total_tables,
                "relationshipCount": total_relationships,
                "dimensionCount": len(dimensions),
                "factCount": len(facts),
                "nodeCount": len(nodes),
                "edgeCount": len(edges)
            },
            "nodes": nodes,
            "edges": edges,
            "dimensions": dimensions,
            "facts": facts
        }

def get_datamodel_service(job_id: Optional[str] = None):
    """
    Get or create the service instance for a specific job_id.
    
    Priority for job_id resolution:
    1. Explicit parameter
    2. Context-local storage (set by datamodel_context)
    3. Environment variables (MODEL_ID, JOB_ID)
    4. First available job directory
    
    Args:
        job_id: Job ID (optional - will use context-local or environment)
    
    Returns:
        KGDataModelService instance for the job
        
    Raises:
        ValueError: If no job_id can be determined
        KGConnectionError: If KG service is unreachable (propagated from service methods)
    """
    global _service_instances
    
    # Determine job_id with priority
    actual_job_id = (
        job_id or                          # 1. Explicit parameter
        _get_current_job_id() or          # 2. Context-local (from context manager)
        os.getenv("MODEL_ID") or          # 3. Environment variable
        os.getenv("JOB_ID")               # 4. Alternative env var
    )
    
    if not actual_job_id:
        # Fallback: try to find existing job
        base_dir = Path("central_data_models")
        if base_dir.exists():
            job_dirs = [d for d in base_dir.iterdir() if d.is_dir() and (d / "datamodel.json").exists()]
            if job_dirs:
                actual_job_id = job_dirs[0].name
                logger.warning(f"No job_id in context, using first found: {actual_job_id}")
        
        if not actual_job_id:
            raise ValueError(
                "No job_id available. Use datamodel_context(job_id) or set MODEL_ID/JOB_ID."
            )
    
    # Thread-safe singleton creation
    if actual_job_id not in _service_instances:
        with _service_lock:
            # Double-check after acquiring lock
            if actual_job_id not in _service_instances:
                from .kg_datamodel_service import KGDataModelService
                _service_instances[actual_job_id] = KGDataModelService(job_id=actual_job_id)
    
    return _service_instances[actual_job_id]
