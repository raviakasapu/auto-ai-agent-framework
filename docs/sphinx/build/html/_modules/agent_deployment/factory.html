def _load_env_once() -> None:
    global _ENV_LOADED
    if _ENV_LOADED:
        return
    try:
        from dotenv import load_dotenv  # type: ignore
    except Exception:
        _ENV_LOADED = True
        return
    # Candidates: ENV_FILE, sdk/.env (repo-relative), CWD .env
    candidates: List[Path] = []
    env_hint = os.getenv("ENV_FILE")
    if env_hint:
        candidates.append(Path(env_hint))
    # sdk/.env next to this file's parent (sdk/agent_deployment -> sdk)
    candidates.append(Path(__file__).resolve().parents[1] / ".env")
    # current working directory .env
    candidates.append(Path.cwd() / ".env")

    for p in candidates:
        try:
            if p.exists():
                load_dotenv(dotenv_path=str(p), override=False)
                break
        except Exception:
            continue
    _ENV_LOADED = True


def _instantiate_from_registry(type_name: str, params: Dict[str, Any]) -> Any:
    # Aggregate registries into one lookup
    aggregated: Dict[str, Type] = {}
    aggregated.update(PLANNER_REGISTRY)
    aggregated.update(MEMORY_REGISTRY)
    aggregated.update(TOOL_REGISTRY)
    aggregated.update(SUBSCRIBER_REGISTRY)
    aggregated.update(PROMPT_REGISTRY)
    aggregated.update(GATEWAY_REGISTRY)
    # Include dynamically registered extras, if any (resolved at call time)
    extras = globals().get("AgentFactory")
    if extras and hasattr(extras, "_EXTRA_COMPONENTS"):
        aggregated.update(getattr(extras, "_EXTRA_COMPONENTS"))

    cls = aggregated.get(type_name)
    if not cls:
        raise ValueError(f"Unknown component type '{type_name}' in registries")
    try:
        return cls(**(params or {}))
    except TypeError:
        return cls()

def _expand_env_vars(yaml_text: str) -> str:
    return os.path.expandvars(yaml_text)


# ------------------ New declarative schema (v1) ------------------
def _is_v1_schema(doc: Dict[str, Any]) -> bool:
    return isinstance(doc, dict) and "apiVersion" in doc and "kind" in doc and "spec" in doc


def _build_standard_agent_v1(config: Dict[str, Any]) -> Agent:
    metadata = config.get("metadata", {})
    resources_def = config.get("resources", {}) or {}
    spec = config["spec"]

    # Instantiate shareable resources by name
    resources_by_name: Dict[str, Any] = {}
    for category, items in resources_def.items():
        for item in items or []:
            name = item["name"]
            comp_type = item["type"]
            params = item.get("config", {})
            resources_by_name[name] = _instantiate_from_registry(comp_type, params)

    # Planner
    planner_def = spec.get("planner", {})
    planner_type = planner_def.get("type")
    planner_params = dict(planner_def.get("config", {}))
    # Resolve planner references to resources (simple rule: if value is a string and matches a resource name, replace)
    for key, value in list(planner_params.items()):
        if isinstance(value, str) and value in resources_by_name:
            planner_params[key] = resources_by_name[value]
    planner = _instantiate_from_registry(planner_type, planner_params)

    # Memory
    memory_def = spec.get("memory", {})
    memory_type = memory_def.get("type")
    memory_params = memory_def.get("config", {})
    memory = _instantiate_from_registry(memory_type, memory_params)

    # Tools by reference name
    tool_names = spec.get("tools", [])
    tools: List[Any] = []
    for tname in tool_names:
        if tname not in resources_by_name:
            raise ValueError(f"Tool resource '{tname}' not found in resources")
        tools.append(resources_by_name[tname])

    # Events: optional (we allow using legacy events block for now if present)
    event_bus = EventBus()

    return Agent(
        planner=planner,
        memory=memory,
        tools=tools,
        event_bus=event_bus,
        name=metadata.get("name", "Agent"),
        description=metadata.get("description", "An AI agent."),
        version=metadata.get("version", "0.1.0"),
    )


def _build_manager_agent_v1(config: Dict[str, Any]) -> ManagerAgent:
    metadata = config.get("metadata", {})
    resources_def = config.get("resources", {}) or {}
    spec = config["spec"]

    # Instantiate resources
    resources_by_name: Dict[str, Any] = {}
    for category, items in resources_def.items():
        for item in items or []:
            name = item["name"]
            comp_type = item["type"]
            params = item.get("config", {})
            resources_by_name[name] = _instantiate_from_registry(comp_type, params)

    # Planner
    planner_def = spec.get("planner", {})
    planner_type = planner_def.get("type")
    planner_params = dict(planner_def.get("config", {}))
    for key, value in list(planner_params.items()):
        if isinstance(value, str) and value in resources_by_name:
            planner_params[key] = resources_by_name[value]
    planner = _instantiate_from_registry(planner_type, planner_params)

    # Memory
    memory_def = spec.get("memory", {})
    memory_type = memory_def.get("type")
    memory_params = memory_def.get("config", {})
    memory = _instantiate_from_registry(memory_type, memory_params)

    # Workers (recursive)
    workers: List[Any] = []
    for worker_def in spec.get("workers", []) or []:
        worker_path = worker_def.get("agent_config_path")
        if not worker_path:
            continue
        workers.append(AgentFactory.create_from_yaml(worker_path))

    event_bus = EventBus()

    return ManagerAgent(
        planner=planner,
        memory=memory,
        workers=workers,
        event_bus=event_bus,
        name=metadata.get("name", "ManagerAgent"),
        description=metadata.get("description", "An AI manager agent."),
        version=metadata.get("version", "0.1.0"),
    )

    @classmethod
    def create_from_yaml(cls, filepath: str) -> Any:
        # Load environment variables from .env if present (once per process)
        _load_env_once()
        path = Path(filepath)
        if not path.exists():
            raise FileNotFoundError(f"Config file not found: {filepath}")
        text = path.read_text(encoding="utf-8")
        text = _expand_env_vars(text)
        raw = yaml.safe_load(text) or {}

        if not _is_v1_schema(raw):
            raise ValueError("Unsupported agent config schema. Expected v1 (apiVersion/kind/spec).")

        kind = raw.get("kind")
        # Prefer custom-registered builders first
        if kind in cls._AGENT_BUILDERS:
            return cls._AGENT_BUILDERS[kind](raw)
        if kind == "Agent":
            return _build_standard_agent_v1(raw)
        if kind == "ManagerAgent":
            return _build_manager_agent_v1(raw)
        raise ValueError(f"No builder registered for agent kind: '{kind}'")
