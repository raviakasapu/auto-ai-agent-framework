def _load_env_once() -> None:
    global _ENV_LOADED
    if _ENV_LOADED:
        return
    try:
        from dotenv import load_dotenv  # type: ignore
    except Exception:
        _ENV_LOADED = True
        return
    # Candidates: ENV_FILE, sdk/.env (repo-relative), CWD .env
    candidates: List[Path] = []
    env_hint = os.getenv("ENV_FILE")
    if env_hint:
        candidates.append(Path(env_hint))
    # sdk/.env next to this file's parent (sdk/deployment -> sdk)
    candidates.append(Path(__file__).resolve().parents[1] / ".env")
    # current working directory .env
    candidates.append(Path.cwd() / ".env")

    for p in candidates:
        try:
            if p.exists():
                load_dotenv(dotenv_path=str(p), override=False)
                break
        except Exception:
            continue
    _ENV_LOADED = True

def _configure_logging_once() -> None:
    global _LOGGING_CONFIGURED
    if _LOGGING_CONFIGURED:
        return
    candidates: List[Path] = []
    config_hint = os.getenv("AGENT_LOGGING_CONFIG")
    if config_hint:
        candidates.append(Path(config_hint))
    default_path = Path(__file__).resolve().parents[1] / "configs" / "logging" / "default.yaml"
    candidates.append(default_path)
    for candidate in candidates:
        try:
            if candidate.exists():
                text = candidate.read_text(encoding="utf-8")
                payload = yaml.safe_load(text) or {}
                level = payload.get("level")
                if level:
                    from agent_framework.logging import set_level
                    set_level(str(level))
                fmt = payload.get("format")
                if fmt:
                    os.environ["AGENT_LOG_FORMAT"] = str(fmt)
                router = payload.get("router")
                if isinstance(router, dict) and "log_details" in router:
                    flag = "true" if router.get("log_details") else "false"
                    os.environ["AGENT_LOG_ROUTER_DETAILS"] = flag
                extra_env = payload.get("environ")
                if isinstance(extra_env, dict):
                    for key, value in extra_env.items():
                        os.environ.setdefault(str(key), str(value))
                break
        except Exception:
            continue
    _LOGGING_CONFIGURED = True

def resolve_config_path(filepath: str) -> Path:
    candidate = Path(filepath)
    if candidate.exists():
        return candidate
    if candidate.is_absolute():
        raise FileNotFoundError(f"Config file not found: {filepath}")

    sdk_root = Path(__file__).resolve().parents[1]
    search_roots: List[Path] = [Path.cwd(), sdk_root]
    if candidate.parent != Path('.'):
        subdirs = [Path('.')]
    else:
        subdirs = [Path('.'), Path('configs/agents'), Path('agent_configs')]

    for base in search_roots:
        for sub in subdirs:
            resolved = (base / sub / candidate).resolve()
            if resolved.exists():
                return resolved

    raise FileNotFoundError(f"Config file not found: {filepath}")



def _instantiate_from_registry(type_name: str, params: Dict[str, Any]) -> Any:
    # Aggregate registries into one lookup
    aggregated: Dict[str, Type] = {}
    aggregated.update(PLANNER_REGISTRY)
    aggregated.update(MEMORY_REGISTRY)
    aggregated.update(TOOL_REGISTRY)
    aggregated.update(SUBSCRIBER_REGISTRY)
    aggregated.update(PROMPT_REGISTRY)
    aggregated.update(GATEWAY_REGISTRY)
    aggregated.update(POLICY_REGISTRY)
    # Include dynamically registered extras, if any (resolved at call time)
    extras = globals().get("AgentFactory")
    if extras and hasattr(extras, "_EXTRA_COMPONENTS"):
        aggregated.update(getattr(extras, "_EXTRA_COMPONENTS"))

    cls = aggregated.get(type_name)
    if not cls:
        raise ValueError(f"Unknown component type '{type_name}' in registries")
    try:
        return cls(**(params or {}))
    except TypeError:
        return cls()

def _expand_env_vars(yaml_text: str) -> str:
    """
    Expand environment variables in YAML text.
    Supports both ${VAR} and ${VAR:-default} syntax.
    
    Resolution order (for thread-safety):
    1. Request context (thread-local) - for per-request vars like JOB_ID
    2. Environment variables - for global config like OPENAI_API_KEY
    """
    import re
    from agent_framework.services.request_context import get_from_context
    
    def replacer(match):
        var_expr = match.group(1)
        # Check if it has a default value (VAR:-default)
        if ':-' in var_expr:
            var_name, default_value = var_expr.split(':-', 1)
            var_name = var_name.strip()
            default_value = default_value.strip()
            
            # Try request context first (thread-safe), then env vars, then default
            value = get_from_context(var_name)
            if value is None:
                value = os.getenv(var_name)
            return value if value is not None else default_value
        else:
            # Simple variable reference
            var_name = var_expr.strip()
            
            # Try request context first (thread-safe), then env vars
            value = get_from_context(var_name)
            if value is None:
                value = os.getenv(var_name)
            return value if value is not None else match.group(0)  # Return original if not found
    
    # Match ${VAR} or ${VAR:-default}
    pattern = r'\$\{([^}]+)\}'
    return re.sub(pattern, replacer, yaml_text)


def _load_policies(spec: Dict[str, Any], resources_by_name: Dict[str, Any]) -> Dict[str, Any]:
    """Load policies from spec, supporting presets and explicit definitions."""
    policies_spec = spec.get("policies", {})
    
    # Check for preset
    if "$preset" in policies_spec:
        preset_name = policies_spec["$preset"]
        policies = get_preset(preset_name)
        
        # Allow overriding preset values
        for key, value in policies_spec.items():
            if key != "$preset":
                # If value is a dict with type/config, instantiate it
                if isinstance(value, dict) and "type" in value:
                    p_type: str = value["type"]
                    policy_params = value.get("config", {})
                    # Resolve resource references
                    for param_key, param_value in list(policy_params.items()):
                        if isinstance(param_value, str) and param_value in resources_by_name:
                            policy_params[param_key] = resources_by_name[param_value]
                    policies[key] = _instantiate_from_registry(p_type, policy_params)
                else:
                    # Direct override
                    policies[key] = value
        
        return policies
    
    # Explicit policy definitions
    policies = {}
    for policy_name, policy_def in policies_spec.items():
        if not isinstance(policy_def, dict) or "type" not in policy_def:
            raise ValueError(f"Policy '{policy_name}' must have 'type' field")
        
        policy_type: str = policy_def["type"]
        policy_params = policy_def.get("config", {})
        
        # Resolve resource references in policy params
        for key, value in list(policy_params.items()):
            if isinstance(value, str) and value in resources_by_name:
                policy_params[key] = resources_by_name[value]
        
        policy = _instantiate_from_registry(policy_type, policy_params)
        policies[policy_name] = policy
    
    return policies


def _build_standard_agent_v2(config: Dict[str, Any]) -> Agent:
    """Build Agent from v2 schema with policies."""
    metadata = config.get("metadata", {})
    resources_def = config.get("resources", {}) or {}
    spec = config["spec"]

    # Instantiate shareable resources by name
    resources_by_name: Dict[str, Any] = {}
    for category, items in resources_def.items():
        for item in items or []:
            name = item["name"]
            comp_type = item["type"]
            params = item.get("config", {})
            resources_by_name[name] = _instantiate_from_registry(comp_type, params)

    # Load policies (REQUIRED in v2)
    policies = _load_policies(spec, resources_by_name)
    
    # Validate required policies
    required_policies = ["completion", "termination", "loop_prevention"]
    for policy_name in required_policies:
        if policy_name not in policies:
            raise ValueError(f"Required policy '{policy_name}' not provided in v2 schema.")

    # Planner
    planner_def = spec.get("planner", {})
    planner_type = planner_def.get("type")
    planner_params = dict(planner_def.get("config", {}))
    RESOURCE_PARAM_KEYS = {"inference_gateway", "gateway"}
    for key, value in list(planner_params.items()):
        if key in RESOURCE_PARAM_KEYS and isinstance(value, str) and value in resources_by_name:
            planner_params[key] = resources_by_name[value]
    planner = _instantiate_from_registry(planner_type, planner_params)

    # Memory
    memory_def = spec.get("memory", {})
    memory_type = memory_def.get("type")
    memory_params = memory_def.get("config", {})
    memory = _instantiate_from_registry(memory_type, memory_params)

    # Tools by reference name
    tool_names = spec.get("tools", [])
    tools: List[Any] = []
    for tname in tool_names:
        if tname not in resources_by_name:
            raise ValueError(f"Tool resource '{tname}' not found in resources")
        tools.append(resources_by_name[tname])

    # Give planners visibility into actual tool objects
    try:
        if hasattr(planner, "configure_tools"):
            tool_map = {getattr(t, "name", f"tool{i}"): t for i, t in enumerate(tools)}
            planner.configure_tools(tool_map)
    except Exception:
        pass

    # Events & subscribers
    event_bus = EventBus()
    subscriber_names = spec.get("subscribers", [])
    for sname in subscriber_names:
        subscriber = resources_by_name.get(sname)
        if not subscriber:
            raise ValueError(f"Subscriber resource '{sname}' not found in resources")
        event_bus.subscribe(subscriber)

    return Agent(
        planner=planner,
        memory=memory,
        tools=tools,
        policies=policies,  # REQUIRED in v2
        event_bus=event_bus,
        name=metadata.get("name", "Agent"),
        description=metadata.get("description", "An AI agent."),
        version=metadata.get("version", "2.0.0"),
    )


def _build_manager_agent_v2(config: Dict[str, Any]) -> ManagerAgent:
    """Build ManagerAgent from v2 schema with policies."""
    metadata = config.get("metadata", {})
    resources_def = config.get("resources", {}) or {}
    spec = config["spec"]

    # Instantiate resources
    resources_by_name: Dict[str, Any] = {}
    for category, items in resources_def.items():
        for item in items or []:
            name = item["name"]
            comp_type = item["type"]
            params = item.get("config", {})
            resources_by_name[name] = _instantiate_from_registry(comp_type, params)

    # Load policies (REQUIRED in v2)
    policies = _load_policies(spec, resources_by_name)
    
    # Validate required policies for Manager
    required_policies = ["completion", "follow_up", "loop_prevention"]
    for policy_name in required_policies:
        if policy_name not in policies:
            raise ValueError(f"Required policy '{policy_name}' not provided for ManagerAgent in v2 schema.")

    # Planner
    planner_def = spec.get("planner", {})
    planner_type = planner_def.get("type")
    planner_params = dict(planner_def.get("config", {}))
    RESOURCE_PARAM_KEYS = {"inference_gateway", "gateway"}
    for key, value in list(planner_params.items()):
        if key in RESOURCE_PARAM_KEYS and isinstance(value, str) and value in resources_by_name:
            planner_params[key] = resources_by_name[value]
    planner = _instantiate_from_registry(planner_type, planner_params)

    # Memory
    memory_def = spec.get("memory", {})
    memory_type = memory_def.get("type")
    memory_params = memory_def.get("config", {})
    memory = _instantiate_from_registry(memory_type, memory_params)

    # Manager tools (optional)
    tool_names = spec.get("tools", [])
    manager_tools: List[Any] = []
    for tname in tool_names:
        if tname not in resources_by_name:
            raise ValueError(f"Manager tool resource '{tname}' not found in resources")
        manager_tools.append(resources_by_name[tname])

    # Give planner visibility into manager tools
    try:
        if hasattr(planner, "configure_tools"):
            tool_map = {getattr(t, "name", f"tool{i}"): t for i, t in enumerate(manager_tools)}
            planner.configure_tools(tool_map)
    except Exception:
        pass

    # Workers (recursive)
    workers_spec = spec.get("workers", []) or []
    workers_map: Dict[str, Any] = {}
    idx = 0
    for worker_def in workers_spec:
        if isinstance(worker_def, str):
            worker_path = worker_def
            name_hint = None
        else:
            worker_path = worker_def.get("agent_config_path")
            name_hint = worker_def.get("name")
        if not worker_path:
            continue
        # Recursive call to build worker agents
        agent_obj = AgentFactory.create_from_yaml(worker_path)
        key = name_hint or getattr(agent_obj, "name", None) or f"worker{idx}"
        idx += 1
        while key in workers_map:
            key = f"{key}_{idx}"
            idx += 1
        workers_map[key] = agent_obj

    event_bus = EventBus()
    subscriber_names = spec.get("subscribers", [])
    for sname in subscriber_names:
        subscriber = resources_by_name.get(sname)
        if not subscriber:
            raise ValueError(f"Subscriber resource '{sname}' not found in resources")
        event_bus.subscribe(subscriber)

    # Optional synthesis configuration
    synthesis_gateway = None
    synthesis_prompt = None
    synthesis_config = spec.get("synthesis", {})
    if synthesis_config:
        gateway_name = synthesis_config.get("gateway")
        if gateway_name and gateway_name in resources_by_name:
            synthesis_gateway = resources_by_name[gateway_name]
        synthesis_prompt = synthesis_config.get("prompt")
    
    # Optional synthesizer agent (separate agent for detailed synthesis)
    synthesizer_agent = None
    synthesizer_agent_path = spec.get("synthesizer_agent")
    if synthesizer_agent_path:
        try:
            synthesizer_agent = AgentFactory.create_from_yaml(synthesizer_agent_path)
        except Exception as e:
            import logging
            logger = logging.getLogger(__name__)
            logger.warning(f"Failed to load synthesizer agent from {synthesizer_agent_path}: {e}")
            synthesizer_agent = None
    
    return ManagerAgent(
        planner=planner,
        memory=memory,
        workers=workers_map,
        policies=policies,  # REQUIRED in v2
        tools=manager_tools,
        event_bus=event_bus,
        name=metadata.get("name", "ManagerAgent"),
        description=metadata.get("description", "An AI manager agent."),
        version=metadata.get("version", "2.0.0"),
        synthesis_gateway=synthesis_gateway,
        synthesis_prompt=synthesis_prompt,
        synthesizer_agent=synthesizer_agent,
    )

    @classmethod
    def create_from_yaml(cls, filepath: str) -> Any:
        """Create agent from YAML config. Only supports v2 schema (apiVersion: agent.framework/v2)."""
        # Load environment variables from .env if present (once per process)
        _load_env_once()
        _configure_logging_once()
        path = resolve_config_path(filepath)
        text = path.read_text(encoding="utf-8")
        text = _expand_env_vars(text)
        raw = yaml.safe_load(text) or {}

        api_version = raw.get("apiVersion", "")
        
        # Only support v2 schema
        if api_version != "agent.framework/v2":
            raise ValueError(
                f"Unsupported agent config schema. Expected 'apiVersion: agent.framework/v2', "
                f"got '{api_version}'. All configs must use v2 schema."
            )

        kind = raw.get("kind")
        if kind == "Agent":
            return _build_standard_agent_v2(raw)
        if kind == "ManagerAgent":
            return _build_manager_agent_v2(raw)
        
        # Check for custom builders
        if kind in cls._AGENT_BUILDERS:
            return cls._AGENT_BUILDERS[kind](raw)
        
        raise ValueError(f"No builder registered for agent kind: '{kind}' in v2 schema")
